def read_file(filename):
    with open(filename, 'r') as file:
        return file.readlines()

def process_commands(commands):
    for command in commands:
        command = command.strip()
        if command:
            print(f"Would execute: {command}")
            # Here you would typically process the command
            # But since we can't execute it, we just print what it would do

def main():
    commands_file = "commands.txt"  # Name of the first file containing commands
    commands = read_file(commands_file)
    process_commands(commands)
